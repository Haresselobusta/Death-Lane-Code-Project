using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class GarageHit : MonoBehaviour
{
    public GameObject garagePanel;
    public GarageUI garageUI;
    private void OnTriggerEnter2D(Collider2D collision)
    {
        if (collision.CompareTag("Player"))
        {
            Debug.Log("VEHICLE REACHED GARAGE!");

            if (AttachmentManager.Instance != null)
            {
                AttachmentManager.Instance.StopAutoHeal();
                AttachmentManager.Instance.hasAutoHealAttachment = false;
            }

            GarageUI.lastSavedSpeed = GameManager.Instance.gameSpeed; // restore the game speed

            GameManager.Instance.isGameOver = true; //shows the garage
            GameManager.Instance.gameSpeed = 0; //Stop the game movement

            if (DistanceTracker.Instance != null)
                DistanceTracker.Instance.PauseCounting();

            garagePanel.SetActive(true);

            garageUI.ShowGarage();
        }
    }
}
