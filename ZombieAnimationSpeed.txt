using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class ZombieAnimationSpeed : MonoBehaviour
{
    private Animator anim;

    public float animationMultiplier = 0.1f;

    private void Awake()
    {
        anim = GetComponent<Animator>();
    }

    private void Update()
    {
        float gameSpeed = GameManager.Instance.gameSpeed;

        anim.speed = 1f + (gameSpeed * animationMultiplier);
    }
}
